head(2byte):0xff,0xfd	
type(1byte):x
addr(2byte):x(低位),x(高位)
data(4byte):x,x,x,x	
crc(2byte):x(低位),x(高位)

crc(2byte):x(低位),x(高位)

head(2byte):0xff,0xff
type(1byte):x
addr(2byte):x(低位),x(高位)
